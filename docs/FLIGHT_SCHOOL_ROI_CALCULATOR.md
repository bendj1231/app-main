## ROI Calculator Tool

### Interactive Calculator

```javascript
// Flight School ROI Calculator
function calculateROI(students, referralRate, conversionRate, commission) {
  const referred = students * referralRate;
  const premium = referred * conversionRate;
  const revenue = premium * commission;
  const investment = 0;
  const profit = revenue - investment;
  const roi = investment === 0 ? 'Infinite' : (profit / investment) * 100;
  
  return {
    students,
    referralRate,
    conversionRate,
    commission,
    referred,
    premium,
    revenue,
    investment,
    profit,
    roi
  };
}

// Example usage
const result = calculateROI(150, 0.5, 0.4, 20);
console.log(result);
// Output: {
//   students: 150,
//   referralRate: 0.5,
//   conversionRate: 0.4,
//   commission: 20,
//   referred: 75,
//   premium: 30,
//   revenue: 600,
//   investment: 0,
//   profit: 600,
//   roi: 'Infinite'
// }
```
